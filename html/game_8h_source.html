171  void Move(){

175

177  s[i].x=s[i-1].x;

179  s[i].d=s[i-1].d;
180  }
184  try {
185  Exception e(10);

187  s[0].y+=1;

190  throw e;

192  }
193  if (dir==left) {
194  s[0].x-=1;
195  s[0].d = left;
196  if (s[0].x<0) {
197  throw e;
198  }
199  }

201  s[0].x+=1;
202  s[0].d = right;
203  if (s[0].x==N) {
204  throw e;
205  }
206  }
207  if (dir==up) {
208  s[0].y-=1;
209  s[0].d = up;
210  if (s[0].y<0) {
211  throw e;
212  }

214  } catch (Exception n) {
218  if (s[0].x==N) s[0].x=0;
219  if (s[0].x<0) s[0].x=N-1;
220  if (s[0].y==M) s[0].y=0;
221  if (s[0].y<0) s[0].y=M-1;
222  }
223 
227  if ((s[0].x==c.x) && (s[0].y==c.y)){
228  num++;

230  c.x=rand() % N;
